## üìñ Project Overview

**LocalFreelance AI** is an open-source, AI-powered portfolio directory built for local freelancers ‚Äî photographers, videographers, tutors, technicians, designers, and more. It offers:

- A **Freelancer Dashboard** to create and manage professional portfolios
- A **User Dashboard** to discover, search, and connect with local freelancers
- An **AI-powered semantic search bar** using Google Gemini that understands natural language queries
- **Zero commission fees** ‚Äî direct client-to-freelancer interaction
- **Fair AI-based ranking** ‚Äî no paid promotions, purely merit and activity-based

---

## ‚ú® Key Features

### For Freelancers

- Secure registration and login (JWT-based)
- Free portfolio creation with bio, skills, service categories, and location
- Upload profile photos and portfolio media
- Set availability, pricing range, and contact preferences
- View incoming client inquiries from the dashboard
- Track profile views and engagement analytics

### For Users / Clients

- Browse and search freelancers by category, location, and skill
- AI-powered natural language search (e.g., _"find a wedding photographer near downtown under $200"_)
- View detailed freelancer profiles and portfolios
- Send direct messages or contact requests to freelancers
- Save/bookmark favorite freelancers
- Rate and review after project completion

### AI Features (Google Gemini)

- Semantic search understanding ‚Äî queries parsed with Gemini to extract intent, service type, location, and budget
- Smart freelancer recommendations ranked by: portfolio quality, activity score, rating, and locality
- Auto-tagging of portfolio uploads (image/video recognition via Gemini Vision)
- AI-generated profile completeness suggestions for freelancers

# accounts/models.py

class CustomUser(AbstractBaseUser, PermissionsMixin):
    """Base user model for both freelancers and clients."""
    email           = models.EmailField(unique=True)
    username        = models.CharField(max_length=50, unique=True)
    role            = models.CharField(choices=[('freelancer', 'Freelancer'), ('client', 'Client')])
    is_active       = models.BooleanField(default=True)
    is_verified     = models.BooleanField(default=False)
    date_joined     = models.DateTimeField(auto_now_add=True)

    USERNAME_FIELD  = 'email'


class FreelancerProfile(models.Model):
    """Extended profile for freelancers."""
    user            = models.OneToOneField(CustomUser, on_delete=models.CASCADE)
    display_name    = models.CharField(max_length=100)
    tagline         = models.CharField(max_length=200)
    bio             = models.TextField()
    profile_photo   = models.URLField(blank=True)       # Cloudinary URL
    categories      = models.ManyToManyField('portfolio.Category')
    skills          = models.ManyToManyField('portfolio.Skill')
    city            = models.CharField(max_length=100)
    state           = models.CharField(max_length=100)
    country         = models.CharField(max_length=100)
    latitude        = models.DecimalField(max_digits=9, decimal_places=6, null=True)
    longitude       = models.DecimalField(max_digits=9, decimal_places=6, null=True)
    price_min       = models.PositiveIntegerField(null=True)
    price_max       = models.PositiveIntegerField(null=True)
    availability    = models.CharField(choices=[
                        ('available', 'Available'),
                        ('busy', 'Busy'),
                        ('offline', 'Offline')
                      ])
    ai_tags         = models.JSONField(default=list)    # Auto-generated by Gemini
    activity_score  = models.FloatField(default=0.0)   # Computed by ranking engine
    profile_views   = models.PositiveIntegerField(default=0)
    created_at      = models.DateTimeField(auto_now_add=True)
    updated_at      = models.DateTimeField(auto_now=True)


class ClientProfile(models.Model):
    """Extended profile for clients."""
    user            = models.OneToOneField(CustomUser, on_delete=models.CASCADE)
    full_name       = models.CharField(max_length=100)
    profile_photo   = models.URLField(blank=True)
    city            = models.CharField(max_length=100)
    bookmarks       = models.ManyToManyField(FreelancerProfile, blank=True)
```

# portfolio/models.py

class Category(models.Model):
    """Service categories: Photography, Tutoring, etc."""
    name    = models.CharField(max_length=100, unique=True)
    slug    = models.SlugField(unique=True)
    icon    = models.CharField(max_length=50)   # CSS icon class or emoji


class Skill(models.Model):
    name     = models.CharField(max_length=100, unique=True)
    category = models.ForeignKey(Category, on_delete=models.SET_NULL, null=True)


class Portfolio(models.Model):
    """A freelancer's main portfolio container."""
    freelancer      = models.OneToOneField('accounts.FreelancerProfile', on_delete=models.CASCADE)
    title           = models.CharField(max_length=200)
    description     = models.TextField()
    is_published    = models.BooleanField(default=False)
    completeness    = models.IntegerField(default=0)    # 0‚Äì100 score
    created_at      = models.DateTimeField(auto_now_add=True)
    updated_at      = models.DateTimeField(auto_now=True)


class PortfolioItem(models.Model):
    """Individual work samples in a portfolio."""
    portfolio       = models.ForeignKey(Portfolio, related_name='items', on_delete=models.CASCADE)
    title           = models.CharField(max_length=200)
    description     = models.TextField()
    media_url       = models.URLField()             # Cloudinary URL
    media_type      = models.CharField(choices=[
                        ('image', 'Image'),
                        ('video', 'Video'),
                        ('document', 'Document')
                      ])
    ai_tags         = models.JSONField(default=list)  # Gemini Vision auto-tags
    is_featured     = models.BooleanField(default=False)
    created_at      = models.DateTimeField(auto_now_add=True)
```

### 2. Fair AI Ranking Algorithm

```python
# apps/search/ranking.py

def compute_activity_score(freelancer_profile) -> float:
    """
    Fair ranking score ‚Äî no paid promotions ever.

    Components:
    - Portfolio completeness  30%
    - Average rating          25%
    - Activity recency        20%
    - Profile views trend     15%
    - Review count            10%
    """
    completeness_score = (freelancer_profile.portfolio.completeness / 100) * 30
    rating_score       = (freelancer_profile.avg_rating / 5.0) * 25
    activity_score     = compute_recency_score(freelancer_profile) * 20
    views_score        = compute_views_trend(freelancer_profile) * 15
    review_score       = min(freelancer_profile.review_count / 10, 1.0) * 10

    return round(
        completeness_score + rating_score + activity_score + views_score + review_score, 2
    )
```

### Frontend JWT Manager (`static/js/auth.js`)

```javascript
const AuthManager = {
  setTokens(access, refresh) {
    localStorage.setItem("access_token", access);
    localStorage.setItem("refresh_token", refresh);
  },

  getAccessToken() {
    return localStorage.getItem("access_token");
  },

  async refreshToken() {
    const refresh = localStorage.getItem("refresh_token");
    const res = await fetch("/api/auth/token/refresh/", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ refresh }),
    });
    if (res.ok) {
      const data = await res.json();
      localStorage.setItem("access_token", data.access);
      return data.access;
    }
    this.logout();
  },

  logout() {
    localStorage.removeItem("access_token");
    localStorage.removeItem("refresh_token");
    window.location.href = "/login/";
  },
};
```

const AISearchBar = {
  debounceTimer: null,

  async search(query) {
    if (query.length < 3) return;
    clearTimeout(this.debounceTimer);

    this.debounceTimer = setTimeout(async () => {
      const res = await fetch(`/api/search/?q=${encodeURIComponent(query)}`, {

      const data = await res.json();
